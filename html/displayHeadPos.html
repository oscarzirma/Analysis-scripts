function displayHeadPos(headPosition)

n=length(headPosition);

for i = 1 : n %generate cell matrices for each dimension, removing any zeros
    hp=cell2mat(headPosition(i,1));
    t=hp(:,1)';
    r=find(t~=0);
    n=length(r);
    T(i)={t(r)};
    x=hp(:,2)';
    X(i)={x(r)};
    x=hp(:,3)';
    Y(i)={x(r)};
    x=hp(:,4)';
    Z(i)={x(r)};
end

figure;
subplot(131);hold on
subplot(132);hold on
subplot(133);hold on

for i=1:n
    subplot(131)
    plot(cell2mat(T(i)),cell2mat(X(i)));
    subplot(132)
    plot(cell2mat(T(i)),cell2mat(Y(i)),'g');
    subplot(133)
    plot(cell2mat(T(i)),cell2mat(Z(i)),'r');
end
